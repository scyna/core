context account;

type Account{
    id: int64
    name: string
    email: string
    password: string
    dob: string
    avatar: string
}

event AccountCreated{
    id: int64
    name: string
    email: string
    password: string
}

event AccountUpdated{
    id: int64
    name: string
    email: string
}

command RegisterAccount{
    request {
        email: string `required, is_email`
        name: string
        password: string
    }
    response {
        id: int64
    }
}

command CreateAccount{
    request: Account validate{
        email `required, is_email`
        name `required, len(1,100)`
    }
    response {
        id: int64
    }
}

query GetAccount{
    request {
        email: string
    }
    response: Account 
}

error{
    ACCOUNT_NOT_EXISTS = 100 `Account Not Exists`
    ACCOUNT_EXISTS = 100 `Account Exists`
}
